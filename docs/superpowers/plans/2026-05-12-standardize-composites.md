# Standardize v2ecoli Composites Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the three hand-rolled `make_*_composite()` factory functions and matched `generate*.py` document builders with three `@composite_generator`-decorated functions registered via `pbg-superpowers`, rename the partitioned architecture to "baseline", consolidate shared infra into `v2ecoli/core.py`, and expose a top-level `v2ecoli.build_composite(name, **kwargs) -> Composite` helper.

**Architecture:** Each architecture becomes a function under `v2ecoli/composites/{baseline,departitioned,reconciled}.py` decorated with `@composite_generator(name=..., parameters=...)`. Shared infrastructure (`build_core`, `load_cache_bundle`, `save_cache`) moves to `v2ecoli/core.py`. `v2ecoli/__init__.py` imports the composites subpackage to fire the decorators, then exposes `build_composite(name, *, core=None, **kwargs)` that resolves the architecture from the `pbg_superpowers._REGISTRY` and constructs a `Composite`. The six legacy files (`composite*.py` + `generate*.py`) are deleted in this PR; every caller is updated in lockstep.

**Tech Stack:** Python 3.12, process-bigraph, bigraph-schema, pbg-superpowers (new runtime dep), pytest.

**Branch:** `standardize-composites` (already cut from `main`, spec committed as `61b7701`).

**Spec:** `docs/superpowers/specs/2026-05-12-standardize-composites-design.md`.

```python
"""Shared infrastructure for v2ecoli composites.

Provides the bigraph-schema ``core`` (with ECOLI_TYPES registered) and the
cache-loading/saving helpers used by every architecture's
``@composite_generator`` function in ``v2ecoli/composites/``.
"""

from __future__ import annotations

import copy
import functools
import os
from typing import Any

import dill
from bigraph_schema import allocate_core

from v2ecoli.cache import load_initial_state, save_initial_state, save_json
from v2ecoli.library.cache_version import (
    StaleCacheError,
    verify_cache_version,
    write_cache_version,
)
# Import at module load so the shared pint UnitRegistry has
# nucleotide/amino_acid/count defined before any dill.load hydrates
# a Quantity whose unit string references those names.
from v2ecoli.library.unit_bridge import rebind_cache_quantities  # noqa: F401
from v2ecoli.types import ECOLI_TYPES


__all__ = [
    "build_core",
    "load_cache_bundle",
    "save_cache",
    "StaleCacheError",
]


def build_core():
    """Create and configure a bigraph-schema core with ecoli types."""
    core = allocate_core()
    core.register_types(ECOLI_TYPES)
    return core


@functools.lru_cache(maxsize=4)
def _load_cache_bundle_cached(cache_dir: str) -> dict[str, Any]:
    """LRU-cached cache loader keyed by directory path."""
    return _load_cache_bundle_uncached(cache_dir)


def load_cache_bundle(cache_dir: str) -> dict[str, Any]:
    """Load the ParCa cache bundle from ``cache_dir``.

    Returns a dict with keys ``configs``, ``unique_names``,
    ``dry_mass_inc_dict``, and ``initial_state`` (and any other entries that
    ``save_cache`` placed in ``sim_data_cache.dill``).
    """
    verify_cache_version(cache_dir)
    return _load_cache_bundle_cached(cache_dir)


def _load_cache_bundle_uncached(cache_dir: str) -> dict[str, Any]:
    """Read the cache files. Used by the LRU cache wrapper above."""
    dill_path = os.path.join(cache_dir, "sim_data_cache.dill")
    with open(dill_path, "rb") as f:
        bundle = dill.load(f)
    # initial_state is saved separately as JSON for inspectability.
    initial_state_path = os.path.join(cache_dir, "initial_state.json")
    if os.path.exists(initial_state_path):
        bundle["initial_state"] = load_initial_state(initial_state_path)
    return bundle


def save_cache(sim_data_path: str, cache_dir: str = "out/cache", seed: int = 0) -> None:
    """Build the cache bundle from ``sim_data_path`` and write it under
    ``cache_dir`` as ``sim_data_cache.dill`` + ``initial_state.json``.

    Also writes ``cache_version.json`` with the fingerprint over the
    INPUT_FILES tuple in ``v2ecoli/library/cache_version.py``.
    """
    # The body of this function is migrated verbatim from
    # v2ecoli/composite.py:save_cache. Copy the body of save_cache from
    # the existing composite.py into here, replacing references to
    # _build_core() with build_core(), and _load_cache_bundle_cached()
    # with _load_cache_bundle_cached() (same name, now in this module).
    raise NotImplementedError(
        "Migrate the body of save_cache from v2ecoli/composite.py into this stub."
    )
```

```python
"""v2ecoli — whole-cell E. coli model on process-bigraph."""

from __future__ import annotations

from typing import Any

from process_bigraph import Composite
from pbg_superpowers.composite_generator import _REGISTRY, build_generator

from v2ecoli.core import build_core
from v2ecoli import composites  # noqa: F401 — forces @composite_generator decorators to fire


def build_composite(
    name: str,
    *,
    core: Any = None,
    **kwargs: Any,
) -> Composite:
    """Build a Composite by architecture name.

    Parameters
    ----------
    name:
        One of ``"baseline"``, ``"departitioned"``, ``"reconciled"``.
    core:
        Optional bigraph-schema core. If omitted, a fresh one is built via
        ``v2ecoli.core.build_core()``.
    **kwargs:
        Passed through to the generator's declared parameters (currently
        ``seed`` and ``cache_dir`` for all three architectures).

    Raises
    ------
    ValueError
        If ``name`` does not match any registered architecture, or matches more
        than one, or if ``kwargs`` contains an unknown parameter name.
    """
    if core is None:
        core = build_core()
    matches = [e for e in _REGISTRY.values() if e.name == name]
    if not matches:
        available = sorted({e.name for e in _REGISTRY.values()})
        raise ValueError(
            f"unknown composite architecture {name!r}; available: {available}"
        )
    if len(matches) > 1:
        raise ValueError(
            f"ambiguous architecture name {name!r}; multiple generators registered: "
            f"{[e.id for e in matches]}"
        )
    doc = build_generator(matches[0], overrides=kwargs, core=core)
    return Composite(doc, core=core)


__all__ = ["build_composite", "build_core"]
```
